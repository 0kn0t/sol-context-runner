```solidity
/// @inheritdoc IMorphoBase
function setOwner(address newOwner) external onlyOwner() {
    require(newOwner != owner, ErrorsLib.ALREADY_SET);
    owner = newOwner;
    emit EventsLib.SetOwner(newOwner);
}
```
